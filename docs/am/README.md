 ## рѕѕрѕЮріЋ Mobiletto?

 ### рІ░рѕЁріЊ рѕЂріЉ рІерѕ╗рїГ рѕўрЅєрѕѕрЇірІФ!
 рІерЅ░рѕѕрІФрІЕ рІерІ░рѕўріЊ рѕЏріерѕЏрЅ╗ ріарЅЁрѕФрЅбрІјрЅй рЅ░рі│рІЃріЮ рІФрѕЇрѕєріЉ ріцрЇњріарІГрІјрЅй ріарѕЈрЅИрІЇрЇб рѕѕ "S3 рЅ░рі│рѕЃріЮріљрЅх" рІерѕџрїЦрѕЕрЅхрѕЮ ріЦріЋрі│ріЋ
 рЇѕрѕірїБрІі рїарЅБрІГ ріарѕІрЅИрІЇрЇб

 рѕѕрѕўрЅ░рїЇрЅарѕфрІФрІј рІерЅ░рѕѕрІе рІерѕЏріерѕЏрЅ╗ ріарЅЁрѕФрЅбріЋ рѕ▓рѕўрѕГрїА рЅарЅђрїЦрЅ│ рІѕрІ░ ріцрЇњріарІГрІј ріерїѕрЅА рѕўрЅ░рїЇрЅарѕфрІФрІј
 ріарѕЂріЋ рЅарІџрѕЁ ріарїѕрѕЇрїЇрѕјрЅх рѕІрІГ рІерЅ░рѕўрѕ░рѕерЅ░ ріљрІЇ. рїірІю ріЦрІФрѕѕрЇѕ рѕ▓рѕёрІх ріЦріЊ рі«рІх рѕ▓рїарѕФрЅђрѕЮ, рѕ╗рї«рЅйріЋ рѕўрЅђрІерѕГ рІГрѕєріЊрѕЇ
 ріЦрІерїерѕўрѕе рІерѕЏрІГрѕёрІх. рІѕрІ░ ріарѕхрІ░рѕ│рЅй рІерѕ╗рїГ рѕўрЅєрѕѕрЇірІФ рІЊрѕѕрѕЮ ріЦріЋрі│ріЋ рЅарІ░рѕЁріЊ рѕўрїА!

 рѕърЅБрІГрѕЇрЅХ рІГрѕЁріЋ рЅйрїЇрѕГ рѕѕрѕўрЇЇрЅ│рЅх рІерЅ░ріљрІ░рЇѕ ріљрІЇрЇб рѕўрЅ░рїЇрЅарѕфрІФрІјріЋ рІѕрІ░ mobiletto's API рі«рІх рЅарѕЏрІхрѕерїЇ рЅарЅђрѕІрѕЅ рІГрЅйрѕІрѕЅрЇб
 рІерѕЏріерѕЏрЅ╗ ріарЅЁрѕФрЅбрІјрЅйріЋ рІГрЅђрІГрѕЕ ріЦріЊ рІерѕўрЅ░рїЇрЅарѕфрІФрІј рѕЏріерѕЏрЅ╗ ріЋрЅЦрѕГрЅЦрѕГ рЅ░рѕўрѕ│рѕ│рІГ рЅБрѕЁрѕф ріЦріЋрІ│рѕѕрІЇ рІГрІѕрЅЂрЇб

 ### рѕ░рЇі рѕЎріерѕФ
 рѕЂрѕЅрѕЮ ріарѕйріерѕГріФрѕфрІјрЅй рѕѕрЅ░рѕўрѕ│рѕ│рІГ рЅБрѕЁрѕф рІерѕџрЇѕрЅ░ріЉрЅх рѕѕріЦрІФріЋрІ│ріЋрІ▒ рѕЙрЇїрѕГ ріе60 рЅарѕІрІГ рЇѕрЅ░ріЊрІјрЅй ріљрІЇрЇб
 рѕЂрѕЅріЋрѕЮ ріарѕйріерѕГріФрѕфрІјрЅй рЅаріЦрІФріЋрІ│ріЋрІ▒ рїЦрѕЮрѕерЅх ріЦріЋрѕъріГрѕФрѕѕріЋрЇА-
 * рѕЮрѕхрїарѕФрЇА рѕЂрѕѕрЅ▒рѕЮ ріљрЅЁрЅ░рІІрѕЇ ріЦріЊ рЅ░рѕ░ріЊріГрѕѕрІІрѕЇрЇб
 * Redis рѕўрѕИрїјрїФрЇА рѕЂрѕѕрЅ▒рѕЮ ріљрЅЁрЅ░рІІрѕЇ ріЦріЊ рЅ░рѕ░ріЊріГрѕѕрІІрѕЇрЇб

 рІГрѕЁ ріаріФрѕёрІх рІерЅхріЏрІЇріЋрѕЮ рѕ╣рЇїрѕГ рЅбрїарЅђрѕЎ рѕърЅБрІГрѕЇрЅХ рЅ░рѕўрѕ│рѕ│рІГ рЅБрѕЁрѕф ріЦріЋрІ░рѕџріќрѕерІЇ рІеріаріЦрѕЮрѕ« рѕ░рѕІрѕЮ рІГрѕ░рїаріЊрѕЇрЇб
 ріЦріЊ рѕўрѕИрїјрїФ ріЦріЊ/рІѕрІГрѕЮ рѕЮрѕхрїарѕФріЋ рЅбрІФріљрЅЃрЅЂрЇб

 ### рІеріарѕйріерѕГріФрѕф рІхрїІрЇЇ
 рІеріарѕЂріЉ рІерѕърЅБрІГрѕЇрЅХ рѕЏріерѕЏрЅ╗ ріљрїѓрІјрЅйрЇА-
 * `s3` : Amazon S3
 * `b2` B2
 * `local` рЇА рІеріаріФрЅБрЅб рІерЇІрІГрѕЇ рѕхрѕГрІЊрЅх

 *рЅ░рїерѕЏрѕф рІерІ░рѕўріЊ рѕЏріерѕЏрЅ╗ ріарЅЁрѕФрЅбрІјрЅйріЋ рѕѕрѕўрІ░рїѕрЇЇ рІерѕџрІ░рѕерїЅрЅх ріарѕхрЅ░рІІрїйрІќрІјрЅй рЅарїБрѕЮ ріЦріЋрі│ріЋ рІ░рѕЁріЊ рѕўрїА!*

 # рѕўрѕ░рѕерЅ│рІі ріарїарЅЃрЅђрѕЮ

    const { mobiletto } = require('mobiletto')

    // General usage
    const api = await mobiletto(driverName, key, secret, opts)

    // To use 'local' driver:
    // * key: base directory
    // * secret: ignored, can be null
    // * opts object:
    // * readOnly: optional, never change anything on the filesystem; default is false
    // * fileMode: optional, permissions used when creating new files, default is 0600. can be string or integer
    // * dirMode: optional, permissions used when creating new directories, default is 0700. can be string or integer
    const local = await mobiletto('local', '/home/ubuntu/tmp', null, {fileMode: 0o0600, dirMode: '0700'})

    // To use 's3' driver:
    // * key: AWS Access Key ID
    // * secret: AWS Secret Key
    // * opts object:
    // * readOnly: optional, never change anything on the bucket; default is false
    // * bucket: required, name of the S3 bucket
    // * region: optional, the AWS region to communicate with, default is us-east-1
    // * prefix: optional, all read/writes within the S3 bucket will be under this prefix
    // * delimiter: optional, directory delimiter, default is '/' (note: always '/' when encryption is enabled)
    const s3 = await mobiletto('s3', aws_key, aws_secret, {bucket: 'bk', region: 'us-east-1'})

    // To use 'b2' driver:
    // * key: Backblaze Key ID
    // * secret: Backblaze Application Key
    // * opts object:
    // * readOnly: optional, never change anything on the bucket; default is false
    // * bucket: required, the ID (**not the name**) of the B2 bucket
    // * prefix: optional, all read/writes within the B2 bucket will be under this prefix
    // * delimiter: optional, directory delimiter, default is '/' (note: always '/' when encryption is enabled)
    // * partSize: optional, large files will be split into chunks of this size when uploading
    const b3 = await mobiletto('b2', b2_key_id, b2_app_key, {bucket: 'bk', partSize: 10000000})

    // List files
    api.list() // --> returns an array of metadata objects

    // List files recursively
    api.list({ recursive: true })

    // List files in a directory
    const path = 'some/path'
    api.list(path)
    api.list(path, { recursive: true }) // also supports recursive flag

    // Visit files in a directory -- visitor function must be async
    api.list(path, { visitor: myAsyncFunc })
    api.list(path, { visitor: myAsyncFunc, recursive: true })

    // The `list` method throws MobilettoNotFoundError if the path does not exist
    // When you call `safeList` on a non-existent path, it returns an empty array
    api.safeList('/path/that/does/not/exist') // returns []
    
    // Read metadata for a file
    api.metadata(path) // returns metadata object

    // The `metadata` method throws MobilettoNotFoundError if the path does not exist
    // When you call `safeMetadata` on a non-existent path, it returns null
    api.safeMetadata('/tmp/does_not_exist') // returns null
    
    // Read a file
    // Provide a callback that writes the data someplace
    const callback = (chunk) => { ... write chunk somewhere ... }
    api.read(path, callback) // returns count of bytes read

    // Read an entire file at once
    const data = await api.readFile(path) // returns a byte Buffer of the file contents
    
    // Read an entire file at once
    // returns null if an exception would otherwise be thrown
    const bufferOrNull = await bucket.safeReadFile('some/path')

    // Write a file
    // Provide a generator function that yields chunks of data
    const generator = function* () {
      while ( ... more-data-to-return ... ) {
        data = ... load-data ...
        yield data
      }
    }
    local.api(path, generator) // returns count of bytes written

    // Write an entire file at once (convenience method)
    await api.writeFile(path, bufferOrString) // returns count of bytes written

    // Delete a file
    // Quiet param is optional (default false), when set errors will not be thrown if the path does not exist
    // Always returns a value or throws an error.
    // Return value may be a single string of the file removed, or an array of all files removed (driver-dependent)
    const quiet = true
    api.remove(path, {quiet}) // returns single path removed

    // Recursively delete a directory and do it quietly (do not report errors)
    const recursive = true
    const quiet = true
    api.remove(path, {recursive, quiet}) // returns array of paths removed

 ### рѕфрІ▓рѕх рѕўрѕИрїјрїФ ріарІГрЇѕрѕЇрїЅрѕЮ?

 ### рѕўрѕИрїјрїФ рѕўрѕўрѕфрІФ
 **рІерЅ░рѕўрѕ░рїарѕе рѕЏріерѕЏрЅ╗**рЇА рІерЅ░рѕўрѕ░рїарѕе рѕЏріерѕЏрЅ╗ рѕЏріЋрЅарЅЦ/рѕўрЇЃрЇЇ ріерѕўрІ░рЅаріЏрІЇ рЅхріЋрѕй рЅђрѕГрЇІрЇІ ріљрІЇрЇб
 ріљрїѕрѕГ рїЇріЋ рѕЏрІЇрїФрІјрЅйріЋ (ріаріЋрІ│ріЋрІх ріљрїѕрѕ«рЅй рІерѕџрѕарѕЕрЅхріЋ) рѕЏрѕ░рѕх рЅарїБрѕЮ рІЇрІх ріљрІЇрЇб рІе redis рѕўрѕИрїјрїФ рЅарѕўрїарЅђрѕЮ
 рїЅрѕЇрѕЁ рІерѕєріљ рІеріарЇѕрЇЃрЇђрѕЮ рѕўрїерѕўрѕГ рІГрѕ░рїЦрІјрЅ│рѕЇ.

 ріљрЅБрѕфрІЇ рѕўрѕИрїјрїФ рІ░рѕЁріЋріљрЅ▒ рІерЅ░рїарЅарЅђ ріљрІЇрЇБ ріљрїѕрѕГ рїЇріЋ рЅЦрІЎ рІерѕўрЇЃрЇЇ/рІерѕЏрѕхрІѕрїѕрІх рѕхрѕФрІјрЅй ріФрѕЅрІјрЅх рЅарІ░ріЋрЅЦ ріарІГрѕ░рѕФрѕЮрЇб
 рѕЏріЋріЏрІЇрѕЮ рІерѕўрї╗рЇЇ рІѕрІГрѕЮ рІерѕЏрѕхрІѕрїѕрІх ріГрІІріћ рѕўрѕІрІЇріЋ рѕўрѕИрїјрїФ рІІрїІ рІФрїарЇІрІІрѕЇрЇБ рІГрѕЁрѕЮ рЅ░ріерЅ│рІГ ріЋрЅБрЅдрЅй рІФрІФрѕЅрЇб
 рІерЅЁрѕГрЅЦ рїірІю рѕѕрІЇрїдрЅй.

 рѕЮріЋ ріЦрІерЅ░рІ░рѕерїѕ ріљрІЇ? рІерЅ░рѕѕрІе "рІерѕЏрІЇрїФ рїЇрЅцрЅх" (рІ│рІГрѕгріГрЅх) рѕЏрІЇрїФ (рІерЅ░рѕўрѕ░рїарѕе) рЅарІџрІФ рІЇрѕхрїЦ рѕЮріЋ рЇІрІГрѕјрЅй ріЦріЋрІ│рѕЅ рІГріерЅ│рЅ░рѕІрѕЇ
 рѕЏрІЇрїФ (рІерЅђрїЦрЅ│ рѕЏрІЇрїФ)рЇб
 * рІеРђ╣рІЮрѕГрІЮрѕГРђ║ рЅхрІЋрІЏрІЎ `list` рІерѕўрїЇрЅбрІФ рЇІрІГрѕјрЅй рІФріљрЅБрѕЇ рЇБрІерЅ░рІўрѕерІўрѕерІЇріЋ ріЦрІФріЋрІ│ріЋрІ▒ріЋ рѕўріЋрїѕрІх рІ▓ріГрѕфрЇЋрЅх рІФрІ░рѕГрїІрѕЇрЇб ріерІџрІФ рѕѕріЦрІФріЋрІ│ріЋрІ▒ рЇІрІГрѕЇ рѕюрЅ│рІ│рЅ│ рІГрѕўрѕЇрѕ│рѕЇ
 *"рІЮрѕГрІЮрѕГ" рЅхрІЋрІЏрІърЅй рІерЅарѕѕрїа рІЇрїцрЅ│рѕЏ ріарІГрІ░рѕЅрѕЮрЇБрЅарЅ░рѕѕрІГ рЅЦрІЎ рЅЂрїЦрѕГ рІФрѕІрЅИрІЇ рЇІрІГрѕјрЅй `list` рѕЏрІЇрїФрІјрЅй
 * `write` рЅхрІЋрІЏрІЎ рЅаріЦрІФріЋрІ│ріЋрІ▒ рІѕрѕІрїЁ рІерЅђрїЦрЅ│ рѕЏрІЇрїФ рІЇрѕхрїЦ рЅђрїЦрЅ░ріЏ рЇІрІГрѕјрЅйріЋ рІГрїйрЇІрѕЇрЇБ рЅарІерїірІюрІЇ; ріерІџрІФрѕЮ рЇІрІГрѕЅріЋ рІГрїйрЇІрѕЇ
 * `write` рЅхріЦрІЏрІърЅй O(N) рІГрїйрЇІрѕЅрЇБ N = рїЦрѕЇрЅђрЅх рЅарѕЏрІЇрїФ рЅ░рІІрѕерІх
 * `remove` рЅхрІЋрІЏрІЎ рЅ░рїЊрІ│ріЮ рЇІрІГрѕЅріЋ рІФрѕхрІѕрїЇрІ│рѕЇрЇБ ріЦріЊ рІѕрѕІрїЂ рЅБрІХ ріерѕєріљрЇБ рЅарЅ░ріерЅ│рЅ│рІГрЇБ ріерІџрІФрѕЮ рЇІрІГрѕЅріЋ рІФрѕхрІѕрїЇрІ│рѕЇ
 * рІФрѕЇрЅ░рІ░рїІрїѕрѕЎ `remove` рЅхріЦрІЏрІърЅй O(N) ріЋрЅБрЅЦ ріЦріЊ рЅЦрІЎ рѕірѕ░рѕГрІЎ рІерѕџрЅйрѕЅ рѕ▓рѕєріЋ N = рїЦрѕЇрЅђрЅх рЅарѕЏрІЇрїФ рЅ░рІІрѕерІх
 * рЅарЅхрѕЇрѕЇрЅЁ ріЦріЊ рїЦрѕЇрЅЁ рІерЇІрІГрѕЇ рѕхрѕГрІЊрЅХрЅй рѕІрІГ рЅ░рІ░рїІрїІрѕџ `remove` рЅхрІЋрІЏрІърЅй рІЇрІх рѕірѕєріЉ рІГрЅйрѕІрѕЅрЇб

 ріерІ░ріЋрЅаріЏ-рїјріЋ рѕЮрѕхрїарѕФ рЅаріљрЅЃрЇБ рЅарЅ░рѕўрѕ│рїарѕе ріарїѕрѕЇрїІрІГрІј рІЇрѕхрїЦ рѕЎрѕЅ рЅ│рІГріљрЅх рІФрѕѕрІЇ рЅБрѕІрїІрѕФ рѕўрѕєріЉріЋ рѕЇрЅЦ рІГрЅарѕЅ

 рІерЅ░рІѕрѕ░ріљ рїЦрѕерЅхрЇБ рІерѕЏрІЇрїФ рЅ░рІІрѕерІх ріарїарЅЃрѕІрІГ рѕўрІІрЅЁрѕГ рїЦрЅѓрЅ▒ріЋ рІѕрІГрѕЮ рѕЂрѕЅріЋрѕЮ рІФрїЇріЎрЇб
 *рѕЏрѕхрЅ│рІѕрѕ╗рЇА рѕѕрЅ░рѕ╗рѕѕ рІ░рѕЁріЋріљрЅх рЅаріаріЋрЇЃрѕФрІіріљрЅх рїарЇЇрїБрЇІ рѕўрІІрЅЁрѕГ рІГрїарЅђрѕЎрЇб*
 рѕЮрѕхрїарѕФрѕЁріЋ ріЦрѕхріФрѕІрІѕрЅђ рІхрѕерѕх рЅБрѕІрїІрѕФрІЇ рІерѕЏрІЇрїФрІјрЅ╣ріЋ/рІерЇІрІГрѕјрЅйріЋ рѕхрѕЮ ріарІФрІЇрЅђрІЇрѕЮрЇб
 рЅЂрѕЇрЇЇ рІѕрІГрѕЮ рЅарѕїрѕІ рѕўріЋрїѕрІх рѕЮрѕхрїарѕФрІЇріЋ рЅарЅ░рѕ│ріФ рѕЂріћрЅ│ рѕ░ріљрїарЅђрІЇрЇб рІФріћ рѕЂрѕЅрѕЮ рІЇрѕГрѕГрІХрЅй рїарЇЇрЅ░рІІрѕЇ!
